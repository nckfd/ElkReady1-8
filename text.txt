function dbLoad() {
  try {
    var raw = localStorage.getItem(STORAGE_KEY);
    if (raw) db = JSON.parse(raw);
  } catch(e) { db = {}; }
}

function dbSave() {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(db));
    var el = document.getElementById("save-status");
    if (el) {
      el.textContent = "Saved";
      el.style.color = "#7eb8a4";
      setTimeout(function(){ el.textContent = ""; }, 1500);
    }
  } catch(e) {
    var el = document.getElementById("save-status");
    if (el) { el.textContent = "Not saving"; el.style.color = "#c87e7e"; }
  }
}

function dbGet(k)    { return db[k] || ""; }
function dbGetNum(k) { return Number(db[k]) || null; }
function dbGetArr(k) { return db[k] ? db[k].split(",").filter(function(x){ return x.length > 0; }) : []; }

function dbSet(k, v) {
  if (v === "" || v === null || v === undefined) { delete db[k]; }
  else { db[k] = String(v); }
  dbSave();
}

function dbToggle(k, v) {
  var arr = dbGetArr(k);
  var idx = arr.indexOf(v);
  if (idx >= 0) arr.splice(idx, 1);
  else arr.push(v);
  dbSet(k, arr.join(","));
}

function sessionHasLogs(w, sk) {
  var exs = SESSIONS[sk].exercises;
  for (var i = 0; i < exs.length; i++) {

sets += “  S” + s + “=” + (db[wk(w,sk,ex.id,s)] || “–”) + “lb x “ + (db[rk(w,sk,ex.id,s)] || “–”) + “r”;
}
var nm = ex.name + (ex.tempo ? “ [3010]” : “”) + (ex.main ? “ *” : “”);
while (nm.length < 26) nm += “ “;
lines.push(nm + sets.trim());
}

var finNote = db[finNoteKey(w, sk)];
lines.push(””, “FINISHER : “ + fin.l, “         “ + fin.d);
if (finNote && finNote.trim()) lines.push(”         Note: “ + finNote.trim());

lines.push(””, “FEEL”, SEP);
lines.push(“Body Feel  : “ + fl(“bodyFeel”));
lines.push(“Recovery   : “ + fl(“recovery”));
lines.push(“Energy     : “ + fl(“energy”));
var rpe = dbGetNum(feelKey(w, sk, “rpe”));
lines.push(“RPE        : “ + (rpe ? rpe + “/10” : “not rated”));
var sore = dbGetArr(feelKey(w, sk, “sore”));
lines.push(“Soreness   : “ + (sore.length ? sore.join(”, “) : “none”));
var stall = dbGetArr(feelKey(w, sk, “stall”));
lines.push(“Stalling   : “ + (stall.length ? stall.join(”, “) : “none”));

// Accessories
var cnt = getAccCount(w, sk);
if (cnt > 0) {
lines.push(””, “ACCESSORY WORK”, SEP);
for (var ai = 0; ai < cnt; ai++) {
var anm = dbGet(accKey(w,sk,ai,“name”));
if (anm.trim()) {
lines.push(”  “ + anm +
“ - “ + dbGet(accKey(w,sk,ai,“sets”)) + “x” + dbGet(accKey(w,sk,ai,“reps”)) +
(dbGet(accKey(w,sk,ai,“lbs”)) ? “ @ “ + dbGet(accKey(w,sk,ai,“lbs”)) + “lbs” : “”) +
(dbGet(accKey(w,sk,ai,“note”)) ? “ (” + dbGet(accKey(w,sk,ai,“note”)) + “)” : “”));
}
}
}

// Conditioning notes
var hasCondNote = false;
for (var ci = 0; ci < CONDITIONING.length; ci++) {
var cn = db[condKey(w, CONDITIONING[ci].id)];
if (cn && cn.trim()) { hasCondNote = true; break; }
}
if (hasCondNote) {
lines.push(””, “CONDITIONING NOTES”, SEP);
for (var ci2 = 0; ci2 < CONDITIONING.length; ci2++) {
var cn2 = db[condKey(w, CONDITIONING[ci2].id)];
if (cn2 && cn2.trim()) lines.push(CONDITIONING[ci2].label + “: “ + cn2.trim());
}
}

var note = db[noteKey(w, sk)];
if (note && note.trim()) lines.push(””, “NOTES”, SEP, note.trim());
var msg = db[msgKey(w, sk)];
if (msg && msg.trim()) lines.push(””, “FOR COACH”, SEP, msg.trim());
lines.push(””, DIV);
return lines.join(”\n”);
}

function doExport() {
try {
var code = btoa(unescape(encodeURIComponent(JSON.stringify(db))));
document.getElementById(“export-box”).value = code;
navigator.clipboard.writeText(code)
.then(function() { showToast(“Copied - paste into Notes app”); })
.catch(function() { showToast(“Code shown below - copy manually”); });
} catch(e) { showToast(“Export failed”, “err”); }
}

function doImport() {
var raw = document.getElementById(“import-input”).value.trim();
if (!raw) { showToast(“Paste your export code first”, “err”); return; }
try {
db = JSON.parse(decodeURIComponent(escape(atob(raw))));
dbSave();
document.getElementById(“import-input”).value = “”;
showToast(“Data restored!”);
goOverview();
} catch(e) { showToast(“Invalid code - check and try again”, “err”); }
}

function doClear() {
if (!confirm(“Delete ALL data? This cannot be undone. Export first if you want a backup.”)) return;
db = {};
localStorage.removeItem(STORAGE_KEY);
showToast(“All data cleared”);
goOverview();
}
